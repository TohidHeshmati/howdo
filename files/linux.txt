[SYS-INFO] uname -a ;; Display detailed system information, including kernel version, hostname, and architecture.
[SYS-INFO] lsb_release -a ;; Show Linux Standard Base (LSB) and distribution-specific information.
[SYS-INFO] hostnamectl ;; Control the system hostname and related settings.
[SYS-INFO] uptime ;; Show how long the system has been running, number of users, and load averages. #performance
[SYS-INFO] date ;; Print or set the system date and time.
[SYS-INFO] cal ;; Display a calendar.

[FS] ls -la ;; List all files and directories in long format, including hidden ones. #list
[FS] df -h ;; Display free disk space on all mounted filesystems in human-readable format. #disk
[FS] du -sh <path> ;; Estimate file space usage for a directory in a human-readable summary. @ path: Path to the directory (e.g., `/var/log`). #disk
[FS] find <path> -name "<pattern>" ;; Search for files by name within a directory. @ path: The directory to start the search from. @ pattern: The file name pattern (e.g., `*.log`). #search
[FS] find <path> -type f -size +<size> ;; Find files larger than a specific size. @ path: The directory to start searching from. @ size: The size (e.g., `100M` for 100MB). #search #disk
[FS] stat <file> ;; Display file or file system status, including permissions and timestamps. @ file: The file to inspect.
[FS] tree -L <level> ;; Display the directory structure as a tree. @ level: The depth of the tree to display.

[PROCESS] ps aux ;; List all running processes on the system. #list
[PROCESS] top ;; Display a real-time, dynamic view of running processes and system resources. #performance
[PROCESS] htop ;; An interactive process viewer, an alternative to `top`.
[PROCESS] kill <pid> ;; Send a signal to a process to terminate it. @ pid: The Process ID to terminate. #danger
[PROCESS] pkill <process_name> ;; Kill processes by name. @ process_name: The name of the process to kill. #danger
[PROCESS] killall <process_name> ;; Kill all processes with a given name. @ process_name: The name of the process to kill. #danger

[PERM] chmod <mode> <file> ;; Change the permissions of a file or directory. @ mode: The permission mode (e.g., `755` or `u+x`). @ file: The target file or directory.
[PERM] chown <user>:<group> <file> ;; Change the owner and group of a file or directory. @ user: The new owner. @ group: The new group. @ file: The target file or directory.
[PERM] sudo <command> ;; Execute a command with superuser privileges. @ command: The command to run as root.

[TEXT] grep "<pattern>" <file> ;; Search for a pattern in a file. @ pattern: The text or regex to search for. @ file: The file to search within.
[TEXT] grep -r "<pattern>" <path> ;; Recursively search for a pattern in a directory. @ pattern: The text or regex to search for. @ path: The directory to search within.
[TEXT] sed 's/<find>/<replace>/g' <file> ;; Find and replace text in a file using a regular expression. @ find: The pattern to find. @ replace: The replacement text. @ file: The file to process.
[TEXT] awk '{print $<column>}' <file> ;; Process and display text by columns. @ column: The column number to print. @ file: The file to process.
[TEXT] wc -l <file> ;; Count the number of lines in a file. @ file: The file to count.
[TEXT] sort <file> ;; Sort the lines of a text file alphabetically. @ file: The file to sort.
[TEXT] uniq <file> ;; Remove duplicate adjacent lines from a sorted file. @ file: The sorted file to process.

[ARCHIVE] tar -czvf <archive_name>.tar.gz <path> ;; Create a compressed archive. @ archive_name: The name of the archive file. @ path: The file or directory to archive.
[ARCHIVE] tar -xzvf <archive_name>.tar.gz ;; Extract a compressed archive. @ archive_name: The archive file to extract.
[ARCHIVE] zip -r <archive_name>.zip <path> ;; Create a compressed zip archive. @ archive_name: The name of the zip file. @ path: The file or directory to archive.
[ARCHIVE] unzip <archive_name>.zip ;; Extract a zip archive. @ archive_name: The zip file to extract.

[USER] whoami ;; Print the current user's username.
[USER] id <username> ;; Display user and group information for a user. @ username: The user to look up.
[USER] useradd <username> ;; Create a new user account. @ username: The name for the new user.
[USER] userdel <username> ;; Delete a user account. @ username: The user to delete. #danger
[USER] passwd <username> ;; Change the password for a user. @ username: The user whose password to change.

[PACKAGE] apt-get update ;; (Debian/Ubuntu) Refresh the package index.
[PACKAGE] apt-get install <package> ;; (Debian/Ubuntu) Install a new package. @ package: The name of the package to install.
[PACKAGE] apt-get remove <package> ;; (Debian/Ubuntu) Remove a package. @ package: The name of the package to remove.
[PACKAGE] yum install <package> ;; (CentOS/RHEL) Install a new package. @ package: The name of the package to install.
[PACKAGE] yum remove <package> ;; (CentOS/RHEL) Remove a package. @ package: The name of the package to remove.
